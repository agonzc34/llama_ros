73inline std::string random_string(int string_size = 32) {
74 static const std::string str(
75 "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz");
76
77 std::random_device rd;
78 std::mt19937 generator(rd());
79
80 std::string result(string_size, ' ');
81
82 for (int i = 0; i < string_size; ++i) {
83 result[i] = str[generator() % str.size()];
84 }
85
86 return result;
87}
